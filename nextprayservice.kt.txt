package com.example.islamicprayertimings

import android.app.*
import android.content.Intent
import android.os.Build
import android.os.IBinder
import androidx.core.app.NotificationCompat

class NextPrayService : Service() {

    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        createNotificationChannel()
        val notification = NotificationCompat.Builder(this, "PRAYER_CHANNEL")
            .setContentTitle("تطبيق مواقيت الصلاة")
            .setContentText("يتم تحديث مواقيت الصلاة في الخلفية...")
            .setSmallIcon(R.drawable.ic_fajr)
            .build()

        startForeground(1, notification)
        
        // هنا يتم حساب وقت الصلاة القادمة وبرمجة التنبيه
        return START_STICKY
    }

    private fun createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val serviceChannel = NotificationChannel(
                "PRAYER_CHANNEL",
                "Prayer Service Channel",
                NotificationManager.IMPORTANCE_DEFAULT
            )
            val manager = getSystemService(NotificationManager::class.java)
            manager.createNotificationChannel(serviceChannel)
        }
    }

    override fun onBind(intent: Intent?): IBinder? = null
}
